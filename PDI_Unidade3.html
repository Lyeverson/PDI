import cv2
import matplotlib.pyplot as plt
import numpy as np

# Lista para armazenar as imagens processadas
canetas = []

borrachas = []

for i in range(1, 11):
    caminho = f'borracha/borracha{i}.jpg'
    imagem = cv2.imread(caminho)

    if imagem is None:
        print(f'erro ao carregar imagem {caminho}')
        continue
    # Convertendo para escala de cinza
    imagem = cv2.cvtColor(imagem, cv2.COLOR_BGR2GRAY)
    # Aplicando filtro Gaussiano
    imagem_blur = cv2.GaussianBlur(imagem, (5, 5), 0)
    # Aplicando limiarizao
    _, thresh = cv2.threshold(imagem_blur, 125, 255, cv2.THRESH_BINARY_INV)
    kernel = np.ones((15, 15), np.uint8)
    dilatacao = cv2.dilate(thresh, kernel, iterations=5)
    thresh = cv2.erode(dilatacao, kernel, iterations=6) 
    borrachas.append(thresh)  # Armazenar a imagem processada

for i in range(1, 11):  # De 1 a 10
    caminho = f'canetas/caneta{i}.jpg'  
    imagem = cv2.imread(caminho)
    
    if imagem is None:
        print(f"Erro ao carregar a imagem {caminho}")
        continue

    imagem = cv2.cvtColor(imagem, cv2.COLOR_BGR2GRAY)
    imagem_blur = cv2.GaussianBlur(imagem, (5, 5), 0)
    _, thresh = cv2.threshold(imagem_blur, 125, 255, cv2.THRESH_BINARY_INV)
    kernel = np.ones((15, 15), np.uint8)
    dilatacao = cv2.dilate(thresh, kernel, iterations=5)
    thresh = cv2.erode(dilatacao, kernel, iterations=6)
    canetas.append(thresh)  

# Exibir as imagens lado a lado
fig, axs = plt.subplots(1, len(canetas), figsize=(20, 5))  # Configurar subplots

# Exibindo as imagens
for ax, img, idx in zip(axs, canetas, range(1, len(canetas) + 1)):
    ax.imshow(img, cmap='gray')
    ax.set_title(f"Caneta {idx}")
    ax.axis('off')

plt.tight_layout()
plt.show()

# Exibir as imagens lado a lado
fig, axs = plt.subplots(1, len(borrachas), figsize=(20, 5))  # Configurar subplots

# Exibindo as imagens
for ax, img, idx in zip(axs, borrachas, range(1, len(borrachas) + 1)):
    ax.imshow(img, cmap='gray')
    ax.set_title(f"borracha {idx}")
    ax.axis('off')

plt.tight_layout()
plt.show()
